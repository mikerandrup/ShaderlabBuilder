Shader.shader


Shader "GlyphEngine/"
{
    Properties
    {
         _MainTex ("Albedo (RGB)", 2D) = "white" {}

    }
    SubShader
    {


        Pass {
            Tags {

                "RenderType"="Opaque"


                "Queue" = "Transparent"
                "IgnoreProjector" = "True"
                "RenderType" = "Transparent"

            }

            LOD 100

            Blend SrcAlpha OneMinusSrcAlpha


            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma target 3.0
            #include "UnityCG.cginc"


            sampler2D _MainTex;

            struct appdata
            {
            };

            struct v2f
            {
            };

            v2f vert(appdata vertData)
            {
                v2f toFrag;

                return toFrag;
            }

            fixed4 frag(v2f i) : SV_Target
            {
                fixed4 col = fixed4(0.5, 0.5, 0.5, 0.5);

                return col;
            }

            ENDCG

        }



        Tags {

            "RenderType"="Opaque"


            "Queue" = "Transparent"
            "IgnoreProjector" = "True"
            "RenderType" = "Transparent"

        }

         LOD 200

         CGPROGRAM

         #pragma surface surf Standard fullforwardshadows
         #pragma target 3.0
         #include "UnityCG.cginc"


         sampler2D _MainTex;
         struct Input
         {
            float2 uv_MainTex;
         };

         void surf (Input IN, inout SurfaceOutputStandard surfOut)
         {

            fixed4 texVal = tex2D (_MainTex, IN.uv_MainTex);

            surfOut.Albedo = texVal.rgb;
            surfOut.Alpha = texVal.a;

         }
         ENDCG

    }
    FallBack ""
}
